


CREATE TABLE T_HPTL_PATIENT 
(
  REC_NUM VARCHAR2(10) NOT NULL 
, FNAME VARCHAR2(100) NOT NULL 
, LNAME VARCHAR2(100) 
, AGE NUMBER NOT NULL 
, DOB DATE NOT NULL 
, GENDER VARCHAR2(1) NOT NULL 
, PHONE VARCHAR2(10) NOT NULL 
, INFO VARCHAR2(4000) 
, CONSTRAINT T_HPTL_PATIENT_PK PRIMARY KEY 
  (
    REC_NUM 
  )
  ENABLE 
);


CREATE SEQUENCE "SQ_T_HPTL_PATIENT"  MINVALUE 1 MAXVALUE 9999999999999999999999999999 INCREMENT BY 1 START WITH 1 NOCACHE  NOORDER  NOCYCLE ;
